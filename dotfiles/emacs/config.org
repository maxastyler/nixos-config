** General
General lets users define keymaps. Works nicely with evil mode
#+BEGIN_SRC emacs-lisp :results silent
(use-package general
:ensure t ;; install the package if it's not installed already
:after which-key
:config
(general-override-mode 1)

  (defun find-user-init-file ()
    "Edit the `user-init-file', in same window."
    (interactive)
    (find-file user-init-file))
  (defun find-user-init-org-file ()
    "Edit the `user-init-file', in same window."
    (interactive)
    (find-file "~/.emacs.d/config.org"))
  (defun load-user-init-file ()
    "Load the `user-init-file', in same window."
    (interactive)
    (load-file user-init-file))


(general-create-definer tyrant-def ;; make a new general definer, tyrant def
:states '(normal visual insert motion emacs)
:prefix "SPC"
:non-normal-prefix "C-SPC")

(tyrant-def  ;; set the keys in the tyrant def
    ""     nil
    "c"   (general-simulate-key "C-c")
    "h"   (general-simulate-key "C-h")
    "u"   (general-simulate-key "C-u")
    "x"   (general-simulate-key "C-x")
    
    "SPC" 'helm-M-x

    ;; Package manager
    "lp"  'list-packages

    ;; Theme operations
    "t"   '(:ignore t :which-key "themes")
    "tn"  'my/cycle-theme
    "tt"  'load-theme
    "tl"  'load-leuven-theme
    "td"  'load-dichromacy-theme
    
    ;; Finding things operations
    "f"   '(:ignore t :which-key "find")
    "fc"  '(:ignore t :which-key "init file")
    "fci" 'find-user-init-file
    "fcr" 'load-user-init-file
    "fco" 'find-user-init-org-file
    
    ;; Project operations
    "p"   '(:ignore t :which-key "project")
    
    ;; Buffer operations
    "b"   '(:ignore t :which-key "buffer")
    "bn"  'evil-next-buffer
    "bp"  'evil-prev-buffer
    "bd"  'kill-this-buffer
    
    ;; Window operations
    "w"   '(:ignore t :which-key "window")
    "wm"  'maximize-window
    "wv"  'split-window-horizontally
    "ws"  'split-window-vertically
    "wm"  'maximize-window
    "wu"  'winner-undo
    "ww"  'other-window
    "wd"  'delete-window
    "wD"  'delete-other-windows
    
    ;; Quitting
    "q"   '(:ignore t :which-key "quit")
    "qq"  'kill-emacs

)
)

** tex
A package for dealing with latex in emacs. Add a buffer local hook to compile after saving.


#+BEGIN_SRC emacs-lisp :results silent
(use-package tex
:ensure auctex ;; install the package if it's not installed already
:mode ("\\.tex\\'" . TeX-latex-mode)
:config
(setq TeX-auto-save t)
(setq TeX-parse-self t)
(setq TeX-PDF-mode t)
(setq-default TeX-master nil)
(setq TeX-source-correlate-start-server 'synctex)
(setq TeX-view-program-selection '(((output-dvi has-no-display-manager)
  "dvi2tty")
 ((output-dvi style-pstricks)
  "dvips and gv")
 (output-dvi "xdvi")
 (output-pdf "PDF Tools")
 (output-html "xdg-open")))
:init
(add-hook 'LaTeX-mode-hook #'my-LaTeX-mode-hooks)
(add-hook 'TeX-after-compilation-finished-functions #'TeX-revert-document-buffer)
(defun my-LaTeX-mode-hooks ()
  (TeX-source-correlate-mode t)
  (LaTeX-math-mode t)
  (add-hook 'after-save-hook (lambda () (TeX-command-run-all nil)) nil t) ;; save after compilation
)
)
#+END_SRC
